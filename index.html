                    <div class="line-tr" style="margin-top:10px;"><span class="lang-badge">EN</span> Some verbs or adjectives can be placed after a verb to explain the result of an action. The negative form uses 没 before the verb (without 了). Questions often end with (了)没有.</div>

        // --- Core Functions ---

        function playAudio(text) {

            // Remove the "A: " or "B: " part for clear pronunciation

            const cleanText = text.replace(/^[A-Z]:\s*/, "");

            const utterance = new SpeechSynthesisUtterance(cleanText);

            utterance.lang = 'zh-CN';

            utterance.rate = 0.9; // Slightly slower for language learners

            window.speechSynthesis.speak(utterance);

        }


        function switchTab(tabId) {

            document.querySelectorAll('.content-section').forEach(el => el.classList.remove('active'));

            document.querySelectorAll('.tab-btn').forEach(el => el.classList.remove('active'));

            document.getElementById(tabId).classList.add('active');

            event.target.classList.add('active');

            window.scrollTo({ top: 0, behavior: 'smooth' });

        }

        // --- Vocab Test (Meaning & Pinyin) ---

        function renderVocabTest() {

            const container = document.getElementById('vocab-test-container');

            container.innerHTML = '';

            

            // 题库中包含测试意思和测试拼音两种题目

            let allQuestions = [];

            vocabList.forEach(q => {

                allQuestions.push({ type: 'meaning', ...q });

                allQuestions.push({ type: 'pinyin', ...q });

            });

            

            // 随机打乱并选择15道题目

            let questions = allQuestions.sort(() => 0.5 - Math.random()).slice(0, 15);

            

            questions.forEach((q, index) => {

                let options = [];

                let correctOpt = "";

                let instruction = "";


                if (q.type === 'meaning') {

                    instruction = "Ushbu iyeroglifning o'zbekcha tarjimasini toping (Ma'nosini toping):";

                    correctOpt = q.uz;

                    let distractors = vocabList.filter(v => v.ch !== q.ch).sort(() => 0.5 - Math.random()).slice(0, 3).map(v => v.uz);

                    options = [correctOpt, ...distractors].sort(() => 0.5 - Math.random());

                } else {

                    instruction = "Ushbu iyeroglifning pinyinini toping (Pinyinni toping):";

                    correctOpt = q.py;

                    let distractors = vocabList.filter(v => v.ch !== q.ch).sort(() => 0.5 - Math.random()).slice(0, 3).map(v => v.py);

                    options = [correctOpt, ...distractors].sort(() => 0.5 - Math.random());

                }


                let correctIdx = options.indexOf(correctOpt);


                let div = document.createElement('div');

                div.className = 'card';

                div.innerHTML = `

                    <div class="quiz-inst">${index + 1}. ${instruction}</div>

                    <div class="quiz-uz-only">${q.ch}</div>

                    <div class="options">

                        ${options.map((opt, i) => `<button class="option-btn" onclick="checkAnswer(this, ${i === correctIdx})">${opt}</button>`).join('')}

                    </div>

                `;

                container.appendChild(div);

            });

        }

        // --- Generic Quiz Renderer ---

        function renderQuiz(containerId, dataArray, instruction) {

            const container = document.getElementById(containerId);

            container.innerHTML = '';

            

            if (instruction) {

                let instDiv = document.createElement('div');

                instDiv.className = 'quiz-inst';

                instDiv.innerText = instruction;

                container.appendChild(instDiv);

            }


            dataArray.forEach((q, index) => {

                let div = document.createElement('div');

                div.className = 'card';

                div.innerHTML = `

                    <div class="quiz-q">${q.q}</div>

                    <div class="options">

                        ${q.opts.map((opt, i) => `<button class="option-btn" onclick="checkAnswer(this, ${i === q.ans})">${opt}</button>`).join('')}

                    </div>

                `;

                container.appendChild(div);

            });

        }


        // --- Quiz Check Answer Logic ---

        function checkAnswer(btn, isCorrect) {

            let siblings = btn.parentElement.children;

            for(let sib of siblings) {

                sib.disabled = true; // Disable all buttons

            }

            if(isCorrect) {

                btn.classList.add('correct');

            } else {

                btn.classList.add('wrong');

                // Highlight the correct one

                let parent = btn.parentElement;

                let optionBtns = Array.from(parent.children);

                // We stored the answer implicitly via the onClick structure, let's find the correct button text

                // by checking their onclick attribute string

                optionBtns.forEach(b => {

                    if (b.getAttribute("onclick").includes("true")) {

                        b.classList.add('correct');

                    }

                });

            }

        }
